1.Permissões no backend, sempre.
2.scopedQuery obrigatório no repository.
3.BuildTaskDTO(task, perspective) — payload mínimo por perfil.
4.Separação por views/services quando faz sentido.
5.Componentes condicionais, nunca dados condicionais.

8.Documentar antes (roadmap-data.ts + AGENT.md).
9.Modal, card e views espelhados (mesmo DTO, mesmo renderer).
10.Local-only, sem push automático.

1.3 scopedQuery — defense in depth

Toda query passa por este helper. Em dev, panic se chamado sem accountID.

// repository_postgres.go
package tasks

import (
    "context"
    "fmt"
    "strings"
    "github.com/jackc/pgx/v5/pgxpool"
)

type PostgresRepository struct{ pool *pgxpool.Pool }

func NewPostgresRepository(pool *pgxpool.Pool) *PostgresRepository {
    return &PostgresRepository{pool: pool}
}

func (r *PostgresRepository) scopedQuery(accountID string, baseSQL string, args ...any) (string, []any) {
    if strings.TrimSpace(accountID) == "" {
        panic("tasks: scopedQuery called without accountID")
    }
    return baseSQL, append([]any{accountID}, args...)
}

func (r *PostgresRepository) GetTask(ctx context.Context, accountID, taskID string) (Task, error) {
    sql, args := r.scopedQuery(accountID, `
        SELECT id, account_id, board_id, column_id, title, content_html,
               status, priority, due_date, start_date, archived, sort_order,
               created_by_user_id, responsible_user_id, client_account_id,
               version, created_at, updated_at
          FROM tasks.tasks
         WHERE account_id = $1 AND id = $2 AND archived = false
    `, taskID)

    var t Task
    err := r.pool.QueryRow(ctx, sql, args...).Scan(
        &t.ID, &t.AccountID, &t.BoardID, &t.ColumnID, &t.Title, &t.ContentHTML,
        &t.Status, &t.Priority, &t.DueDate, &t.StartDate, &t.Archived, &t.SortOrder,
        &t.CreatedByUserID, &t.ResponsibleUserID, &t.ClientAccountID,
        &t.Version, &t.CreatedAt, &t.UpdatedAt,
    )
    if err != nil {
        return Task{}, fmt.Errorf("tasks.GetTask: %w", err)
    }
    return t, nil
}

// service_dto.go
package tasks

type TaskDTO struct {
    ID          string        `json:"id"`
    BoardID     string        `json:"boardId"`
    Title       string        `json:"title"`
    Status      *string       `json:"status,omitempty"`
    Priority    string        `json:"priority"`
    DueDate     *string       `json:"dueDate,omitempty"`
    StartDate   *string       `json:"startDate,omitempty"`
    Responsible *UserMiniDTO  `json:"responsible,omitempty"`

    // APENAS para agency:
    ClientAccountID *string       `json:"clientAccountId,omitempty"`
    Assignees       []UserMiniDTO `json:"assignees,omitempty"`
    TrackingTotalMs *int64        `json:"trackingTotalMs,omitempty"`

    Version int `json:"version"`
}

func (s *Service) BuildTaskDTO(task Task, perspective Perspective) TaskDTO {
    dto := TaskDTO{
        ID:        task.ID,
        BoardID:   task.BoardID,
        Title:     task.Title,
        Status:    task.Status,
        Priority:  task.Priority,
        DueDate:   formatISO(task.DueDate),
        StartDate: formatISO(task.StartDate),
        Version:   task.Version,
    }
    if task.ResponsibleUserID != nil {
        dto.Responsible = s.resolveUser(task.ResponsibleUserID)
    }
    if perspective == PerspectiveAgency {
        dto.ClientAccountID = task.ClientAccountID
        dto.Assignees = s.resolveAssignees(task.ID)
        dto.TrackingTotalMs = ptrInt64(s.computeTrackingTotal(task.ID))
    }
    // client_viewer não recebe nada além do mínimo.
    return dto
}

func (h *HTTPHandler) RegisterRoutes(mux *http.ServeMux, auth middleware.Auth) {
    mux.Handle("GET /v1/tasks/boards",
        auth.RequireAuth(h.withPermission(PermBoardsView, h.listBoards)))

    mux.Handle("POST /v1/tasks/boards",
        auth.RequireAuth(h.withPermission(PermBoardsManage, h.createBoard)))

    mux.Handle("GET /v1/tasks/{taskId}",
        auth.RequireAuth(h.withPermission(PermTasksView, h.getTask)))
}

func (h *HTTPHandler) withPermission(key string, next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        ac := accessContextFrom(r.Context())
        if !ac.Has(key) {
            httpapi.WriteError(w, http.StatusForbidden, "forbidden")
            return
        }
        next(w, r)
    }
}

func (h *HTTPHandler) getTask(w http.ResponseWriter, r *http.Request) {
    ac := accessContextFrom(r.Context())
    id := r.PathValue("taskId")

    task, err := h.service.GetTask(r.Context(), ac, id)
    if errors.Is(err, ErrTaskNotFound) {
        httpapi.WriteError(w, http.StatusNotFound, "not_found") // 404, não 403
        return
    }
    if err != nil {
        httpapi.WriteError(w, http.StatusInternalServerError, "internal")
        return
    }
    httpapi.WriteJSON(w, http.StatusOK, h.service.BuildTaskDTO(task, ac.Perspective))
}

func (s *Service) audit(ctx context.Context, ac AccessContext,

        AccountID:    ac.AccountID,
        UserID:       &ac.UserID,
        Action:       action,
        ResourceType: resourceType,
        ResourceID:   resourceID,
        Before:       before,
        After:        after,
    })
}

func (s *Service) UpdateTask(ctx context.Context, ac AccessContext,
    in UpdateTaskInput) (Task, error) {

    before, err := s.repo.GetTask(ctx, ac.AccountID, in.ID)
    if err != nil { return Task{}, err }
    if before.Version != in.ExpectedVersion {
        return Task{}, ErrVersionConflict
    }

    after, err := s.repo.UpdateTask(ctx, ac.AccountID, in)
    if err != nil { return Task{}, err }

    s.audit(ctx, ac, "task.updated", "task", after.ID, before, after)
    s.publisher.PublishTaskEvent(ctx, TaskEvent{
        Type: "task.updated", AccountID: ac.AccountID,
        BoardID: after.BoardID, TaskID: after.ID, Version: after.Version,
    })
    return after, nil
}

    const task = findTask(taskId)
    const before = { columnId: task.columnId, sortOrder: task.sortOrder }
    // optimistic
    task.columnId = columnId
    task.sortOrder = sortOrder

    try {
      const updated = await $api<Task>(`/v1/tasks/${taskId}/move`, {
        method: 'POST',
        body: { columnId, sortOrder },
      })
      mergeTask(updated)
    } catch (err) {
      task.columnId = before.columnId
      task.sortOrder = before.sortOrder
      throw err
    }
  }

  function applyRealtimeEvent(evt: TaskEvent) {
    if (evt.type === 'task.updated') refetchTask(evt.taskId)
  }

  return { boards, activeBoardId, tasksByBoard, fetchBoards, moveTask, applyRealtimeEvent }
})

func (s *Service) StartTracking(ctx context.Context, ac AccessContext, taskID string) (TimeEntry, error) {
    return s.repo.Tx(ctx, func(tx Querier) (TimeEntry, error) {
        if err := s.assertCanTrack(ctx, tx, ac, taskID); err != nil {

            TaskID: taskID, UserID: ac.UserID,

        if err != nil { return TimeEntry{}, err }

        s.publisher.PublishTaskEvent(ctx, TaskEvent{
            Type: "task.time_started",
            AccountID: ac.AccountID, TaskID: taskID,
            ActorUserID: ac.UserID, SavedAt: entry.StartedAt,
        })
        return entry, nil
    })
}

// realtime/presence.go
type presenceEntry struct {
    UserID, DisplayName, AvatarPath string
    ConnectedAt   time.Time
    LastHeartbeat time.Time
    FieldLocks    map[string]string // fieldKey → lockId
}

type PresenceStore struct {
    mu      sync.RWMutex
    byTopic map[string]map[string]*presenceEntry
}

func (p *PresenceStore) Heartbeat(topic, userID string) {
    p.mu.Lock(); defer p.mu.Unlock()
    if entry, ok := p.byTopic[topic][userID]; ok {
        entry.LastHeartbeat = time.Now()
    }
}

func (p *PresenceStore) sweepLoop() {
    t := time.NewTicker(5 * time.Second)
    for range t.C {
        cutoff := time.Now().Add(-30 * time.Second)
        p.mu.Lock()
        for topic, users := range p.byTopic {
            for uid, e := range users {
                if e.LastHeartbeat.Before(cutoff) {
                    delete(users, uid)
                    p.publish(topic, "presence.user_left", uid)
                }
            }
        }
        p.mu.Unlock()
    }
}

Validação ao criar view

func (s *Service) CreateView(ctx context.Context, ac AccessContext,
    in CreateViewInput) (View, error) {

    board, err := s.repo.GetBoard(ctx, ac.AccountID, in.BoardID)
    if err != nil { return View{}, err }

    desc := viewDescriptors[in.Type]
    for _, reqType := range desc.RequiredFieldTypes {
        if !hasFieldOfType(board, in.Config, reqType) {
            return View{}, fmt.Errorf(
                "%w: view %s requer field tipo %s",
                ErrViewConfigInvalid, in.Type, reqType,
            )
        }
    }
    return s.repo.InsertView(ctx, in)
}
